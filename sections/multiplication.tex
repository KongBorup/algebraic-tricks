\section{Multiplication}
\begin{center}
    \begin{longtable}{lcr}
        $a \cdot b \pm a \cdot c = a(b \pm c)$
        &
        $a_1 \cdot a_2\ \cdots\ \cdot a_n = a^n$
        &
        $n \cdot a = a_1 + a_2\ \cdots\ + a_n$
        \\
        $a \cdot 1 = a$
        &
        $a \cdot 0 = 0$
        &
        $-a \cdot (-b) = a \cdot b$
        \\
        $-a \cdot b = a \cdot (-b)$
    \end{longtable}
\end{center}
